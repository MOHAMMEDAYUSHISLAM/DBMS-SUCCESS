CREATE TABLE BOOKS (
    book_id INT PRIMARY KEY,
    title VARCHAR(100),
    category VARCHAR(50),
    price DECIMAL(10,2)
);

CREATE TABLE MEMBERS (
    member_id INT PRIMARY KEY,
    name VARCHAR(100),
    join_date DATE
);
CREATE TABLE DOCTORS (
    doc_id INT PRIMARY KEY,
    name VARCHAR(100),
    specialization VARCHAR(50),
    salary DECIMAL(10,2)
);

CREATE TABLE PATIENTS (
    patient_id INT PRIMARY KEY,
    name VARCHAR(100),
    age INT,
    disease VARCHAR(100)
);
